// --- [အရေးကြီး] --- // မင်းရဲ့ လက်ရှိ Script အောက်ဆုံးမှာ ဒီ Function တွေကို ထည့်လိုက်ပါ။ // အရင်က ရှိပြီးသား script တွေကို ဒီအသစ်တွေနဲ့ ပြန်အစားထိုးလိုက်ပါ။ function renderImageViewer() { el.imageViewer.innerHTML = ''; queue.forEach(item => { const wrapper = document.createElement('div'); wrapper.className = "w-full border border-zinc-800 p-1 bg-black"; const img = document.createElement('img'); img.src = item.url; img.className = "w-full"; wrapper.appendChild(img); el.imageViewer.appendChild(wrapper); }); } // --- Typesetting System (မင်းလိုချင်တဲ့ Canvas Clipping Logic) --- async function runTypeset(jsonString) { const data = JSON.parse(jsonString); const viewer = el.imageViewer; viewer.innerHTML = ''; // Viewer ကို ရှင်းမယ် for(let item of queue) { const img = new Image(); img.src = item.url; await img.decode(); const canvas = document.createElement('canvas'); canvas.width = img.naturalWidth; canvas.height = img.naturalHeight; const ctx = canvas.getContext('2d'); ctx.drawImage(img, 0, 0); // Clipping logic data.coordinates.forEach((box, i) => { ctx.save(); ctx.beginPath(); ctx.rect(box.x, box.y, box.w, box.h); ctx.clip(); // Bubble အတွင်းမှာပဲ ဆွဲမယ် ctx.fillStyle = "#FFFFFF"; ctx.fillRect(box.x, box.y, box.w, box.h); ctx.fillStyle = "black"; ctx.font = "bold 24px 'Padauk'"; wrapText(ctx, data.translatedBubbles[i], box.x, box.y, box.w, 30); ctx.restore(); }); viewer.appendChild(canvas); } } function wrapText(ctx, text, x, y, maxWidth, lineHeight) { const words = text.split(' '); let line = ''; for(let i = 0; i < words.length; i++) { let testLine = line + words[i] + ' '; if (ctx.measureText(testLine).width > maxWidth && i > 0) { ctx.fillText(line, x, y); line = words[i] + ' '; y += lineHeight; } else { line = testLine; } } ctx.fillText(line, x, y); }
